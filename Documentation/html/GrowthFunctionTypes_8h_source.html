281  GridBasedGrowthFunction(int id, int type, float initTime, float endTime, bool applyToColumnarLayer, bool applyToPeripodialMembrane, int nX, int nY, double*** GrowthMat, double** AngleMat) : GrowthFunctionBase(id, type, initTime, endTime, applyToColumnarLayer, applyToPeripodialMembrane){
288  double aspectRatioThreshold = 1.1;
289  this ->nGridX = nX;
290  this ->nGridY = nY;
291  GrowthMatrix = new double**[(const int) nGridX];
292  xyShearAngleMatrix = new double*[(const int) nGridX];
293  aspectRatioOverThresoldMatrix = new bool*[(const int) nGridX];
294 
295  compatibleAngleEliminated = new bool**[(const int) nGridX];
296  compatibleAngles = new double**[(const int) nGridX];
297  compatibleGrowths = new double***[(const int) nGridX];
298 
299 
300  for (int i=0; i<nGridX; ++i){
301  GrowthMatrix[i] = new double*[(const int) nGridY];
302  xyShearAngleMatrix[i] = new double[(const int) nGridY];
303  aspectRatioOverThresoldMatrix[i] = new bool[(const int) nGridY];
304  compatibleAngleEliminated[i] = new bool*[(const int) nGridY];
305  compatibleAngles[i] = new double*[(const int) nGridY];
306  compatibleGrowths[i] = new double**[(const int) nGridY];
307  for (int j=0; j<nGridY; ++j){
308  GrowthMatrix[i][j] = new double[3];
309  xyShearAngleMatrix[i][j] = AngleMat[i][j]*M_PI/180.0; //converting to radians
310  aspectRatioOverThresoldMatrix[i][j] = true; //setting all aspect ratios to be over threshold
311  for (int k=0; k<3; ++k){
312  //double scaleGrowth[3] = {1.427,2.806,1.0};
313  double scaleGrowth[3] = {1.0,1.0,1.0};
314  //the experimental size at 72 hours should be 210 - 130.
315  //The model results in a size of 160 - 70
316  //The experimental tissue size at 48 hours is 84 - 46
317  //I will scale the rates to give the experimental size:
318  //rx' / rx = ln(210/84)/ln(160/84) = 1.427
319  //ry' / ry = ln(130/46)/ln(70/46) = 2.806
320  GrowthMatrix[i][j][k] = scaleGrowth[k]*GrowthMat[i][j][k];
321  }
322  //cout<<"set the growth matrix for: "<<i<<" "<<j<<endl;
323  compatibleAngleEliminated[i][j] = new bool[4]; //booleans stating if the angle will be added in compatible averaging in for 4 corners
324  compatibleAngles[i][j] = new double[4]; //angles compatible for averaging in for 4 corners
325  compatibleGrowths[i][j] = new double*[4]; //growth rates [rx,ry,rz] compatible for averaging in for 4 corners
326  for (int k=0; k<4; ++k){
327  //cout<<"set default values of compatibleAngleEliminated for: "<<i<<" "<<j<<" "<<k<<endl;
328  compatibleAngleEliminated[i][j][k] = false; //by default all angles are added
329  compatibleAngles[i][j][k] = 0.0; //by default all angles are zero
330  compatibleGrowths[i][j][k]= new double[3]; //growth rates [rx,ry,rz] at index point [i][j], for corner [k]
331  for (int l=0; l<3; ++l){
332  compatibleGrowths[i][j][k][l] = 0.0;
333  }
334  }
335  //cout<<" [i][j]: ["<<i<<"]["<<j<<"], Growth_x&y: "<<GrowthMatrix[i][j][0]<<" "<<GrowthMatrix[i][j][1]<<endl;
336  double AR = GrowthMatrix[i][j][0] / GrowthMatrix[i][j][1];
337  if (AR < 0 ){
338  //some of the growth rates have higher aspect ratio than their nuclei count, for instance, an aspect ratio of 5 with nuclei count 4
339  //then the tissue actually needs to actively shring in one direction while growing in the other.
340  //this will lead to a negative aspect ratio, and this extremely high aspect ratio would be eliminated by the threshold, as it would be less than the positive threshold.
341  AR *=-1.0;
342  }
343  if (AR < 1.0){
344  AR = 1/AR;
345  }
346  if(AR < aspectRatioThreshold){
347  aspectRatioOverThresoldMatrix[i][j] = false;

349  }
350  }
351  }

354  for (int i=0; i<nGridX; ++i){
355  xyShearRotationsMatrix[i] = new gsl_matrix*[(const int) nGridY];

362  gsl_matrix_set(xyShearRotationsMatrix[i][j],0,0, c );
363  gsl_matrix_set(xyShearRotationsMatrix[i][j],0,1, -1.0*s);
364  gsl_matrix_set(xyShearRotationsMatrix[i][j],0,2, 0.0);
365  gsl_matrix_set(xyShearRotationsMatrix[i][j],1,0, s);
366  gsl_matrix_set(xyShearRotationsMatrix[i][j],1,1, c);
367  gsl_matrix_set(xyShearRotationsMatrix[i][j],1,2, 0.0);
368  gsl_matrix_set(xyShearRotationsMatrix[i][j],2,0, 0.0);
369  gsl_matrix_set(xyShearRotationsMatrix[i][j],2,1, 0.0);
370  gsl_matrix_set(xyShearRotationsMatrix[i][j],2,2, 1.0);

434  void preCalculateAnglesForCompatibleAveraging(){
435  for (int IndexX = 0; IndexX<nGridX-1; ++IndexX){
436  for (int IndexY = 0; IndexY<nGridY-1; ++IndexY){
437  //read the growth rates at four corners for this grid point:
438  double growth0[3] = {0.0,0.0,0.0};
439  double growth1[3] = {0.0,0.0,0.0};
440  double growth2[3] = {0.0,0.0,0.0};
441  double growth3[3] = {0.0,0.0,0.0};
442  for (int axis =0; axis<3; axis++){
443  growth0[axis] = getGrowthMatrixElement(IndexX,IndexY,axis);
444  growth1[axis] = getGrowthMatrixElement(IndexX+1,IndexY,axis);
445  growth2[axis] = getGrowthMatrixElement(IndexX,IndexY+1,axis);
446  growth3[axis] = getGrowthMatrixElement(IndexX+1,IndexY+1,axis);
447  }
448  //now reading the angles:
449  double angles[4] = {0.0,0.0,0.0,0.0};
450  //Not summing aspect ratios close to one:
451  //and recording the eliminated angles for ease of use later on:
452  bool AngleEliminated[4] = {false, false, false, false};
453  if (isAspectRatioOverOne( IndexX, IndexY)) {angles[0] = getXyShearAngleMatrixElement(IndexX , IndexY); }
454  else {AngleEliminated[0] = true;}
455  if (isAspectRatioOverOne( IndexX+1, IndexY)) {angles[1] = getXyShearAngleMatrixElement(IndexX+1, IndexY); }
456  else {AngleEliminated[1] = true;}
457  if (isAspectRatioOverOne( IndexX, IndexY+1)) {angles[2] = getXyShearAngleMatrixElement(IndexX, IndexY+1);}
458  else {AngleEliminated[2] = true;}
459  if (isAspectRatioOverOne( IndexX+1, IndexY+1)) {angles[3] = getXyShearAngleMatrixElement(IndexX+1, IndexY+1);}
460  else {AngleEliminated[3] = true;}
461  if ( IndexX == 8 && IndexY ==3){
462  cout<<" angles eliminated: "<<AngleEliminated[0]<<" "<<AngleEliminated[1]<<" "<<AngleEliminated[2]<<" "<<AngleEliminated[3]<<endl;
463  cout<<" angles prior to correction: "<<angles[0]<<" "<<angles[1]<<" "<<angles[2]<<" "<<angles[3]<<endl;
464  }
465  //bring all angles in the interval 0 - 180:
466  for (int i=0; i<4; ++i){
467  while (angles[i]<0){angles[i] += M_PI;}
468  while (angles[i]>M_PI){angles[i] -= M_PI;}
469  }
470  if ( IndexX == 8 && IndexY ==3){
471  cout<<" angles after interval correction: "<<angles[0]<<" "<<angles[1]<<" "<<angles[2]<<" "<<angles[3]<<endl;
472  }
473  //now I am bringing the angles to the vicinity of each other:
474  //selecting the base angle, the first one that was not eliminated will do:
475  bool angleSelected = false;
476  int selectedIndice = 3;
477  double tet = 0;
478  for (int i=0; i<4; ++i){
479  if(!AngleEliminated[i] && !angleSelected){
480  tet = angles[i];
481  angleSelected = true;
482  }
483  if (angleSelected){
484  selectedIndice = i;
485  break;
486  }
487  }
488  if ( IndexX == 8 && IndexY ==3){
489  cout<<" selected indice: "<<selectedIndice<<" tet: "<<tet<<endl;
490  }
491  //selected the first angle, now bring all the rest near this one:
492  //I keep a record if I need to flip the growths:
493  bool flipGrowth[4] = {false, false, false, false};
494  for (int i=selectedIndice; i<4; ++i){
495  double dtet = angles[i] - tet; // angle from selected angle (tet) to current angle (angles[i]).
496  if ( IndexX == 8 && IndexY ==3){
497  cout<<" correcting angle : "<<i<<" dtet: "<<dtet<<endl;
498  }
499  if (dtet>0){
500  //selected tetha is larger than this angle:
501  if (dtet >= M_PI/4.0 && dtet < 3.0*M_PI/4.0 ){
502  //the difference is between 45-135 degrees, I flip the growth, and rotate angle -90 degrees.
503  //The difference comes to the interval (-45) - 45
504  angles[i] = angles[i] - M_PI/2.0;
505  flipGrowth[i] = true;
506  }
507  else if (dtet >= 3.0*M_PI/4.0 && dtet< 5.0*M_PI/4.0 ){
508  //the difference is between 135-225 degrees, I only need to rotate -180 degrees,
509  //no flip necessary, the angle will fall into the interval (-45) - 45
510  angles[i] = angles[i] - M_PI;
511  }
512  else if (dtet >= 5.0*M_PI/4.0 && dtet< 7.0*M_PI/4.0){
513  //the difference is between 225 and 315 degrees, I need to rotate and flip as in the first case.
514  //I need to rotate by -270 degrees to bring to the interval (-45) - 45
515  angles[i] = angles[i] - 6.0*M_PI/4.0;
516  flipGrowth[i] = true;
517  }
518  else if (dtet >= 7.0*M_PI/4.0){
519  //the difference is higher than 315 degrees. I need to rotate the angle by -360 degrees, and
520  //the difference will come into (-45) - 0 (as it cannot be higher than 360. No need to flip
521  angles[i] = angles[i] - 2.0*M_PI;
522  }
523  }
524  else if (dtet<0){
525  //selected tetha is smaller than this angle:
526  if (dtet <= -M_PI/4.0 && dtet > -3.0*M_PI/4.0 ){
527  //the difference is between (-45)- (-135) degrees, I flip the growth, and rotate angle +90 degrees.
528  //The difference comes to the interval (-45) - 45
529  angles[i] = angles[i] + M_PI/2.0;
530  flipGrowth[i] = true;
531  }
532  else if (dtet <= -3.0*M_PI/4.0 && dtet > -5.0*M_PI/4.0 ){
533  //the difference is between (-135) - (-225) degrees, I only need to rotate +180 degrees,
534  //no flip necessary, the angle will fall into the interval (-45) - 45
535  angles[i] = angles[i] + M_PI;
536  }
537  else if (dtet <= -5.0*M_PI/4.0 && dtet > -7.0*M_PI/4.0){
538  //the difference is between (-225) and (-315) degrees, I need to rotate and flip as in the first case.
539  //I need to rotate by +270 degrees to bring to the interval (-45) - 45
540  angles[i] = angles[i] + 6.0*M_PI/4.0;
541  flipGrowth[i] = true;
542  }
543  else if (dtet <= -7.0*M_PI/4.0){
544  //the difference is less than (-315) degrees. I need to rotate the angle by +360 degrees, and
545  //the difference will come into 0 - 45 (as it cannot be lower than (-360). No need to flip
546  angles[i] = angles[i] + 2.0*M_PI;
547  }
548  }
549  }
550  //all angles are brought to the vicinity of the selected base angle;
551  //flipping the growth where necessary:
552  if (flipGrowth[0]){
553  double x = growth0[0];
554  growth0[0] = growth0[1];
555  growth0[1] = x;
556  }
557  if (flipGrowth[1]){
558  double x = growth1[0];
559  growth1[0] = growth1[1];
560  growth1[1] = x;
561  }
562  if (flipGrowth[2]){
563  double x = growth2[0];
564  growth2[0] = growth2[1];
565  growth2[1] = x;
566  }
567  if (flipGrowth[3]){
568  double x = growth3[0];
569  growth3[0] = growth3[1];
